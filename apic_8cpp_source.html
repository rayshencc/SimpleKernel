82auto Apic::StartupAp(uint32_t apic_id, uint64_t ap_code_addr,
83 size_t ap_code_size, uint64_t target_addr) const
84 -> Expected<void> {
85 // 检查参数有效性
86 sk_assert_msg(ap_code_addr != 0 && ap_code_size != 0,
87 "Invalid AP code parameters: addr=0x%llx, size=%zu",
88 ap_code_addr, ap_code_size);
89
90 // 检查目标地址是否对齐到 4KB 边界（SIPI 要求）
91 sk_assert_msg((target_addr & 0xFFF) == 0,
92 "Target address 0x%llx is not aligned to 4KB boundary",
93 target_addr);
94
95 // 检查目标地址是否在有效范围内（实模式可访问）
96 // 1MB 以上的地址在实模式下不可访问
97 sk_assert_msg(target_addr < 0x100000,
98 "Target address 0x%llx exceeds real mode limit (1MB)",
99 target_addr);
100
101 std::memcpy(reinterpret_cast<void*>(target_addr),
102 reinterpret_cast<void*>(ap_code_addr), ap_code_size);
103
104 // 验证复制是否成功
105 if (std::memcmp(reinterpret_cast<const void*>(target_addr),
106 reinterpret_cast<const void*>(ap_code_addr),
107 ap_code_size) != 0) {
108 klog::Err("AP code copy verification failed\n");
109 return std::unexpected(Error(ErrorCode::kApicCodeCopyFailed));
110 }
111
112 // 计算启动向量 (物理地址 / 4096)
113 auto start_vector = static_cast<uint8_t>(target_addr >> 12);
114 // 使用 Local APIC 发送 INIT-SIPI-SIPI 序列
115 local_apic_.WakeupAp(apic_id, start_vector);
116
117 return {};
118}

120void Apic::StartupAllAps(uint64_t ap_code_addr, size_t ap_code_size,
121 uint64_t target_addr) const {
122 sk_assert_msg(ap_code_addr != 0 && ap_code_size != 0,
123 "Invalid AP code parameters: addr=0x%llx, size=%zu",
124 ap_code_addr, ap_code_size);
125
126 // 启动 APIC ID 0 到 cpu_count_-1 的所有处理器
127 // 跳过当前的 BSP (Bootstrap Processor)
128 for (size_t apic_id = 0; apic_id < cpu_count_; apic_id++) {
129 // 跳过当前 BSP
130 if (static_cast<uint32_t>(apic_id) == cpu_io::GetCurrentCoreId()) {
131 continue;
132 }
133 StartupAp(static_cast<uint32_t>(apic_id), ap_code_addr, ap_code_size,
134 target_addr)
135 .or_else([apic_id](Error err) -> Expected<void> {
136 klog::Err("Failed to start AP with APIC ID 0x%x: %s\n", apic_id,
137 err.message());
138 return std::unexpected(err);
139 });
140 }
