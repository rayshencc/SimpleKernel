109void Gic::Gicr::SetUP() const {
110 auto cpuid = cpu_io::GetCurrentCoreId();
111
112 // 将 GICR_CTLR 清零
113 Write(cpuid, kCTLR, 0);
114
115 // The System register interface for the current Security state is enabled.
116 cpu_io::ICC_SRE_EL1::SRE::Set();
117
118 // 允许 Non-secure Group 1 中断
119 Write(cpuid, kIGROUPR0, kIGROUPR0_Set);
120 Write(cpuid, kIGRPMODR0, kIGRPMODR0_Clear);
121
122 // 唤醒 Redistributor
123 // @see
124 // https://developer.arm.com/documentation/ddi0601/2024-12/External-Registers/GICR-WAKER--Redistributor-Wake-Register?lang=en
125 auto waker = Read(cpuid, kWAKER);
126 // Clear the ProcessorSleep bit
127 Write(cpuid, kWAKER, waker & ~kWAKER_ProcessorSleepMASK);
128 // 等待唤醒完成
129 while (Read(cpuid, kWAKER) & kWAKER_ChildrenAsleepMASK) {
130 cpu_io::Pause();
131 }
132}
