122 if (aliased_path != nullptr) {

298 [[nodiscard]] auto GetPlic() const
299 -> Expected<std::tuple<uint64_t, uint64_t, uint32_t, uint32_t>> {
300 sk_assert_msg(fdt_header_ != nullptr, "fdt_header_ is null");
301
302 // 查找 PLIC 节点，依次尝试两个 compatible 字符串
303 auto offset = FindCompatibleNode("sifive,plic-1.0.0");
304 if (!offset.has_value()) {
305 offset = FindCompatibleNode("riscv,plic0");
306 }
307 if (!offset.has_value()) {
308 return std::unexpected(offset.error());
309 }
310
311 int len = 0;
312
313 // 通过 interrupts-extended 字段计算上下文数量
314 const auto* prop = fdt_get_property(fdt_header_, offset.value(),
315 "interrupts-extended", &len);
316 if (prop == nullptr) {
317 return std::unexpected(Error(ErrorCode::kFdtPropertyNotFound));
318 }
319
320 // interrupts-extended 格式: <cpu_phandle interrupt_id> 成对出现
321 // 每两个 uint32_t 值表示一个上下文 (CPU + 模式)
322 uint32_t num_entries = len / sizeof(uint32_t);
323 // 每两个条目表示一个上下文
324 uint32_t context_count = num_entries / 2;
325
326 // 获取 ndev 属性
327 prop = fdt_get_property(fdt_header_, offset.value(), "riscv,ndev", &len);
328 if (prop == nullptr) {
329 return std::unexpected(Error(ErrorCode::kFdtPropertyNotFound));
330 }
331 uint32_t ndev =
332 fdt32_to_cpu(*reinterpret_cast<const uint32_t*>(prop->data));
333
334 // 获取 reg 属性
335 auto reg = GetRegProperty(offset.value());
336 if (!reg.has_value()) {
337 return std::unexpected(reg.error());
338 }
339
340 return std::tuple{reg.value().first, reg.value().second, ndev,
341 context_count};

522 auto validate_id = [this, offset](const char* name,

524 int len = 0;
525 const auto* prop = fdt_get_property(fdt_header_, offset, name, &len);
526 if (prop != nullptr && static_cast<size_t>(len) >= sizeof(uint32_t)) {

529 klog::Debug("PSCI %s function ID: 0x%X\n", name, id);
530 if (id != expected) {
531 klog::Err("PSCI %s function ID mismatch: expected 0x%X, got 0x%X\n",
532 name, expected, id);

534 }
535 }
536 return {};
537 };
538

540 .and_then([&]() { return validate_id("cpu_off", kPsciCpuOffFuncId); })
541 .and_then([&]() {
