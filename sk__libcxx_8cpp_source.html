125extern "C" auto __cxa_guard_acquire(GuardType* guard) -> int {
126 uint64_t expected = 0;
127
128 // 紧凑的 CAS 循环：尝试将 0 -> kInUseMask
129 // compare_exchange_weak 失败时会将当前值写入 expected
130 while (!guard->guard.compare_exchange_weak(expected, GuardType::kInUseMask,
131 std::memory_order_acq_rel,
132 std::memory_order_acquire)) {
133 // CAS 失败，检查失败原因
134 if ((expected & GuardType::kInitializedMask) != 0U) {
135 // 已被其他核心初始化完成
136 return 0;
137 }
138
139 // 其他核心正在初始化中，等待
140 cpu_io::Pause();
141
142 // 重置 expected 以便下次 CAS 尝试
143 // 如果当前值仍有 kInUseMask，CAS 会再次失败并更新 expected
144 expected = 0;
145 }
146
147 // CAS 成功，当前核心负责初始化
148 return 1;
149}
