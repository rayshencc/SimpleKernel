27// Internal helper for string to number conversion
28// Parses magnitude into unsigned long long.
29// handles base detection, whitespace, signs.
30static unsigned long long strtox_main(const char *nptr, char **endptr, int base,
31 int *sign_out, int *overflow) {
32 const char *s = nptr;
33 unsigned long long acc = 0;
34 int c;
35 unsigned long long cutoff;
36 int cutlim;
37 int any = 0;
38 int negative = 0;
39
40 *overflow = 0;
41
42 // Skip whitespace
43 while (isspace(*s)) s++;
44
45 // Check sign
46 if (*s == '-') {
47 negative = 1;
48 s++;
49 } else if (*s == '+') {
50 s++;
51 }
52 if (sign_out) *sign_out = negative;
53
54 // Detect base
55 if ((base == 0 || base == 16) && *s == '0' && (s[1] == 'x' || s[1] == 'X')) {
56 // Hex prefix
57 // We speculatively consume it.
58 if (isxdigit(s[2])) {
59 s += 2;
60 base = 16;
61 } else {
62 // '0x' followed by non-hex.
63 // if base==0, it's octal 0.
64 if (base == 0) base = 8;
65 }
66 }
67 if (base == 0) {
68 base = *s == '0' ? 8 : 10;
69 }
70
71 if (base < 2 || base > 36) {
72 if (endptr) *endptr = (char *)nptr; // Invalid base
73 return 0;
74 }
75
76 cutoff = ULLONG_MAX / (unsigned long long)base;
77 cutlim = ULLONG_MAX % (unsigned long long)base;
78
79 for (;; s++) {
80 c = *s;
81 if (isdigit(c))
82 c -= '0';
83 else if (isalpha(c))
84 c = toupper(c) - 'A' + 10;
85 else
86 break;
87
88 if (c >= base) break;
89
90 if (any < 0 || acc > cutoff || (acc == cutoff && c > cutlim)) {
91 any = -1; // overflow
92 } else {
93 any = 1;
94 acc = acc * base + c;
95 }
96 }
97
98 if (any < 0) {
99 *overflow = 1;
100 acc = ULLONG_MAX;
101 }
102
103 // Set endptr
104 if (endptr) {
105 *endptr = (char *)(any ? s : nptr);
106 }
107 return acc;
108}
