36 __always_inline auto Lock() -> Expected<void> {
37 auto intr_enable = cpu_io::GetInterruptStatus();
38 cpu_io::DisableInterrupt();
39
40 // 先尝试获取锁
41 while (locked_.test_and_set(std::memory_order_acquire)) {
42 // 在等待时检查是否是当前核心持有锁（递归锁检测）
43 if (core_id_.load(std::memory_order_acquire) ==
44 cpu_io::GetCurrentCoreId()) {
45 // 递归锁定，恢复中断状态并返回失败
46 if (intr_enable) {
47 cpu_io::EnableInterrupt();
48 }
49 sk_printf("spinlock %s recursive lock detected.\n", name_);
50 return std::unexpected(Error{ErrorCode::kSpinLockRecursiveLock});
51 }
52 cpu_io::Pause();
53 }
54
55 // 获取锁成功后立即设置 core_id_
56 core_id_.store(cpu_io::GetCurrentCoreId(), std::memory_order_release);
57 saved_intr_enable_ = intr_enable;
58 return {};
59 }

65 __always_inline auto UnLock() -> Expected<void> {
66 if (!IsLockedByCurrentCore()) {
67 sk_printf("spinlock %s IsLockedByCurrentCore == false.\n", name_);
68 return std::unexpected(Error{ErrorCode::kSpinLockNotOwned});
69 }
70
71 // 先重置 core_id_，再释放锁
72 core_id_.store(std::numeric_limits<size_t>::max(),
73 std::memory_order_release);
74 locked_.clear(std::memory_order_release);
75
76 if (saved_intr_enable_) {
77 cpu_io::EnableInterrupt();
78 }
79 return {};
80 }
