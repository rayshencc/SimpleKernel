406 // 创建三个相同权重的任务
407 constexpr size_t kTaskCount = 3;

415 }
416
417 // 模拟多轮调度
418 constexpr int kRounds = 5;
419 for (int round = 0; round < kRounds; ++round) {
420 // 每轮选择所有任务
421 for (size_t i = 0; i < kTaskCount; ++i) {
422 auto* task = scheduler.PickNext();
423 EXPECT_NE(task, nullptr, "Should pick a task");
424
425 // 模拟任务运行一段时间
426 for (int tick = 0; tick < 5; ++tick) {
427 scheduler.OnTick(task);
428 }
429
430 // 将任务重新入队
431 scheduler.Enqueue(task);
432 }
433 }
434
435 // 检查所有任务的 vruntime 应该相近（公平性）
436 uint64_t max_vruntime = 0;
437 uint64_t min_vruntime = UINT64_MAX;
438
439 for (size_t i = 0; i < kTaskCount; ++i) {
440 if (tasks[i]->sched_data.cfs.vruntime > max_vruntime) {
441 max_vruntime = tasks[i]->sched_data.cfs.vruntime;
442 }
443 if (tasks[i]->sched_data.cfs.vruntime < min_vruntime) {
444 min_vruntime = tasks[i]->sched_data.cfs.vruntime;
445 }
446 }
447
448 // vruntime 差异应该较小（表示公平性）
449 uint64_t difference = max_vruntime - min_vruntime;
450 EXPECT_LT(difference, 10000,

452

454 for (size_t i = 0; i < kTaskCount; ++i) {
