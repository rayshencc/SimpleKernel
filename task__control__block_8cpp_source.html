22uint64_t LoadElf(const uint8_t* elf_data, uint64_t* page_table) {
23 // Check ELF magic
24 auto* ehdr = reinterpret_cast<const Elf64_Ehdr*>(elf_data);
25 if (ehdr->e_ident[EI_MAG0] != ELFMAG0 || ehdr->e_ident[EI_MAG1] != ELFMAG1 ||
26 ehdr->e_ident[EI_MAG2] != ELFMAG2 || ehdr->e_ident[EI_MAG3] != ELFMAG3) {
27 klog::Err("Invalid ELF magic\n");
28 return 0;
29 }
30
31 auto* phdr = reinterpret_cast<const Elf64_Phdr*>(elf_data + ehdr->e_phoff);
32 auto& vm = Singleton<VirtualMemory>::GetInstance();
33
34 for (int i = 0; i < ehdr->e_phnum; ++i) {
35 if (phdr[i].p_type != PT_LOAD) continue;
36
37 uintptr_t vaddr = phdr[i].p_vaddr;
38 uintptr_t memsz = phdr[i].p_memsz;
39 uintptr_t filesz = phdr[i].p_filesz;
40 uintptr_t offset = phdr[i].p_offset;
41
42 uint32_t flags = cpu_io::virtual_memory::GetUserPagePermissions(
43 (phdr[i].p_flags & PF_R) != 0, (phdr[i].p_flags & PF_W) != 0,
44 (phdr[i].p_flags & PF_X) != 0);
45
46 uintptr_t start_page = cpu_io::virtual_memory::PageAlign(vaddr);
47 uintptr_t end_page = cpu_io::virtual_memory::PageAlignUp(vaddr + memsz);
48
49 for (uintptr_t page = start_page; page < end_page;
50 page += cpu_io::virtual_memory::kPageSize) {
51 void* p_page = aligned_alloc(cpu_io::virtual_memory::kPageSize,
52 cpu_io::virtual_memory::kPageSize);
53 if (!p_page) {
54 klog::Err("Failed to allocate page for ELF\n");
55 return 0;
56 }
57 std::memset(p_page, 0, cpu_io::virtual_memory::kPageSize);
58
59 // Mapping logic
60 uintptr_t v_start = page;
61 uintptr_t v_end = page + cpu_io::virtual_memory::kPageSize;
62
63 // Map intersection with file data
64 uintptr_t file_start = vaddr;
65 uintptr_t file_end = vaddr + filesz;
66
67 uintptr_t copy_start = std::max(v_start, file_start);
68 uintptr_t copy_end = std::min(v_end, file_end);
69
70 if (copy_end > copy_start) {
71 uintptr_t dst_off = copy_start - v_start;
72 uintptr_t src_off = (copy_start - vaddr) + offset;
73 std::memcpy((uint8_t*)p_page + dst_off, elf_data + src_off,
74 copy_end - copy_start);
75 }
76
77 if (!vm.MapPage(page_table, (void*)page, p_page, flags)) {
78 klog::Err("MapPage failed\n");
79 return 0;
80 }
81 }
82 }
83 return ehdr->e_entry;
84};

88void TaskControlBlock::JoinThreadGroup(TaskControlBlock* leader) {
89 if (!leader || leader == this) {
90 return;
91 }
92
93 // 设置 tgid
94 tgid = leader->tgid;
95
96 // 如果 leader 已经有其他线程，插入到链表头部
97 if (leader->thread_group_next) {
98 thread_group_next = leader->thread_group_next;
99 thread_group_next->thread_group_prev = this;
100 }
101 leader->thread_group_next = this;
