21class MockAllocator {
22 public:
23 static auto GetInstance() -> MockAllocator& {
24 static MockAllocator instance;

26 }

28 auto AlignedAlloc(size_t alignment, size_t size) -> void* {
29 void* ptr = nullptr;
30 if (posix_memalign(&ptr, alignment, size) == 0) {
31 allocated_blocks_[ptr] = size;
32 return ptr;
33 }
34 return nullptr;
35 }
36
37 void Free(void* ptr) {
38 if (ptr == nullptr) {
39 return;
40 }
41 auto it = allocated_blocks_.find(ptr);
42 if (it != allocated_blocks_.end()) {
43 allocated_blocks_.erase(it);
44 free(ptr);
45 }
46 }
47
48 void Reset() {
49 for (auto& [ptr, size] : allocated_blocks_) {
50 free(ptr);
51 }
52 allocated_blocks_.clear();
53 }
54
55 [[nodiscard]] auto GetAllocatedCount() const -> size_t {
56 return allocated_blocks_.size();
57 }
58
59 private:
60 MockAllocator() = default;
61 std::unordered_map<void*, size_t> allocated_blocks_;
62};
