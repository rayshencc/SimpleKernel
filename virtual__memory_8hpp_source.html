81 auto MapMMIO(
82 uint64_t phys_addr, size_t size,
83 uint32_t flags = cpu_io::virtual_memory::GetKernelPagePermissions())
84 -> Expected<void*> {
85 // 计算对齐后的起始和结束页
86 auto start_page = cpu_io::virtual_memory::PageAlign(phys_addr);
87 auto end_page = cpu_io::virtual_memory::PageAlignUp(phys_addr + size);
88
89 // 遍历并映射
90 for (uint64_t addr = start_page; addr < end_page;
91 addr += cpu_io::virtual_memory::kPageSize) {
92 auto result = MapPage(kernel_page_dir_, reinterpret_cast<void*>(addr),
93 reinterpret_cast<void*>(addr), flags);
94 if (!result.has_value()) {
95 return std::unexpected(result.error());
96 }
97 }
98 return reinterpret_cast<void*>(phys_addr);
99 }

109 auto MapPage(void* page_dir, void* virtual_addr, void* physical_addr,
110 uint32_t flags) -> Expected<void> {
111 sk_assert_msg(page_dir != nullptr, "MapPage: page_dir is null");
112
113 // 查找页表项，如果不存在则分配
114 auto pte_result = FindPageTableEntry(page_dir, virtual_addr, true);
115 if (!pte_result.has_value()) {
116 return std::unexpected(Error(ErrorCode::kVmMapFailed));
117 }
118
119 auto pte = pte_result.value();
120
121 // 检查是否已经映射且标志位相同
122 if (cpu_io::virtual_memory::IsPageTableEntryValid(*pte)) {
123 // 如果物理地址和标志位都相同，则认为是重复映射（警告但不失败）
124 auto existing_pa = cpu_io::virtual_memory::PageTableEntryToPhysical(*pte);
125 if (existing_pa == reinterpret_cast<uint64_t>(physical_addr) &&
126 (*pte & ((1ULL << cpu_io::virtual_memory::kPteAttributeBits) - 1)) ==
127 flags) {
128 klog::Debug(
129 "MapPage: duplicate va = %p, pa = 0x%lX, flags = 0x%X, skip\n",
130 virtual_addr, existing_pa, flags);
131 // 重复映射，但不是错误
132 return {};
133 }
134 klog::Warn("MapPage: remap va = %p from pa = 0x%lX to pa = %p\n",
135 virtual_addr, existing_pa, physical_addr);
136 }
137
138 // 设置页表项
139 *pte = cpu_io::virtual_memory::PhysicalToPageTableEntry(
140 reinterpret_cast<uint64_t>(physical_addr), flags);
141 // 刷新 TLB
142 cpu_io::virtual_memory::FlushTLBAll();
143
144 return {};
145 }

274 void RecursiveFreePageTable(uint64_t* table, size_t level, bool free_pages) {
275 if (table == nullptr) {
276 return;
277 }
278
279 // 遍历页表中的所有条目
280 for (size_t i = 0; i < kEntriesPerTable; ++i) {
281 uint64_t pte = table[i];
282 if (!cpu_io::virtual_memory::IsPageTableEntryValid(pte)) {
283 continue;
284 }
285
286 auto pa = cpu_io::virtual_memory::PageTableEntryToPhysical(pte);
287
288 // 如果不是最后一级，递归释放子页表
289 if (level > 0) {
290 RecursiveFreePageTable(reinterpret_cast<uint64_t*>(pa), level - 1,
291 free_pages);
292 } else if (free_pages) {
293 // 最后一级页表，释放物理页
294 aligned_free(reinterpret_cast<void*>(pa));
295 }
296
297 // 清除页表项
298 table[i] = 0;
299 }
300
301 // 如果不是根页表，释放当前页表
302 if (level < cpu_io::virtual_memory::kPageTableLevels - 1) {
303 aligned_free(table);
304 }
305 }

315 auto RecursiveClonePageTable(uint64_t* src_table, uint64_t* dst_table,
316 size_t level, bool copy_mappings)
317 -> Expected<void> {
318 sk_assert_msg(src_table != nullptr,
319 "RecursiveClonePageTable: src_table is null");
320 sk_assert_msg(dst_table != nullptr,
321 "RecursiveClonePageTable: dst_table is null");
322
323 for (size_t i = 0; i < kEntriesPerTable; ++i) {
324 uint64_t src_pte = src_table[i];
325 if (!cpu_io::virtual_memory::IsPageTableEntryValid(src_pte)) {
326 continue;
327 }
328
329 if (level > 0) {
330 // 非最后一级，需要递归复制子页表
331 auto src_pa = cpu_io::virtual_memory::PageTableEntryToPhysical(src_pte);
332 auto* src_next_table = reinterpret_cast<uint64_t*>(src_pa);
333
334 // 分配新的子页表
335 auto* dst_next_table = aligned_alloc(cpu_io::virtual_memory::kPageSize,
336 cpu_io::virtual_memory::kPageSize);
337 if (dst_next_table == nullptr) {
338 return std::unexpected(Error(ErrorCode::kVmAllocationFailed));
339 }
340
341 // 清零新页表
342 std::memset(dst_next_table, 0, cpu_io::virtual_memory::kPageSize);
343
344 // 递归复制子页表
345 auto result = RecursiveClonePageTable(
346 src_next_table, reinterpret_cast<uint64_t*>(dst_next_table),
347 level - 1, copy_mappings);
348 if (!result.has_value()) {
349 aligned_free(dst_next_table);
350 return std::unexpected(result.error());
351 }
352
353 // 设置目标页表项指向新的子页表
354 dst_table[i] = cpu_io::virtual_memory::PhysicalToPageTableEntry(
355 reinterpret_cast<uint64_t>(dst_next_table),
356 cpu_io::virtual_memory::GetTableEntryPermissions());
357 } else {
358 // 最后一级页表
359 if (copy_mappings) {
360 // 直接复制页表项（共享物理页）
361 dst_table[i] = src_pte;
362 }
363 // 如果不复制映射，保持目标页表项为 0
364 }
365 }
366
367 return {};
368 }

377 [[nodiscard]] auto FindPageTableEntry(void* page_dir, void* virtual_addr,
378 bool allocate = false)
379 -> Expected<uint64_t*> {
380 auto* current_table = reinterpret_cast<uint64_t*>(page_dir);
381 auto vaddr = reinterpret_cast<uint64_t>(virtual_addr);
382
383 // 遍历页表层级
384 for (size_t level = cpu_io::virtual_memory::kPageTableLevels - 1; level > 0;
385 --level) {
386 // 获取当前级别的虚拟页号
387 auto vpn = cpu_io::virtual_memory::GetVirtualPageNumber(vaddr, level);
388 auto* pte = &current_table[vpn];
389 if (cpu_io::virtual_memory::IsPageTableEntryValid(*pte)) {
390 // 页表项有效，获取下一级页表
391 current_table = reinterpret_cast<uint64_t*>(
392 cpu_io::virtual_memory::PageTableEntryToPhysical(*pte));
393 } else {
394 // 页表项无效
395 if (allocate) {
396 auto* new_table = aligned_alloc(cpu_io::virtual_memory::kPageSize,
397 cpu_io::virtual_memory::kPageSize);
398 if (new_table == nullptr) {
399 return std::unexpected(Error(ErrorCode::kVmAllocationFailed));
400 }
401 // 清零新页表
402 std::memset(new_table, 0, cpu_io::virtual_memory::kPageSize);
403
404 // 设置中间页表项
405 *pte = cpu_io::virtual_memory::PhysicalToPageTableEntry(
406 reinterpret_cast<uint64_t>(new_table),
407 cpu_io::virtual_memory::GetTableEntryPermissions());
408
409 current_table = reinterpret_cast<uint64_t*>(new_table);
410 } else {
411 return std::unexpected(Error(ErrorCode::kVmPageNotMapped));
412 }
413 }
414 }
415
416 // 返回最底层页表中的页表项
417 auto vpn = cpu_io::virtual_memory::GetVirtualPageNumber(vaddr, 0);
418
419 return &current_table[vpn];
420 }
