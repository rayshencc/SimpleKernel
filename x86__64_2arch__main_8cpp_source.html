152 Singleton<Apic>::GetInstance().InitCurrentCpuLocalApic().or_else(

154 klog::Err("Failed to initialize APIC: %s\n", err.message());
155 while (true) {
156 cpu_io::Pause();

158 return std::unexpected(err);
159 });
